122static inline float EaseOutBounce(float t) {
123 if (t < (1.0f / 2.75f)) {
124 return 7.5625f * t * t;
125 } else if (t < (2.0f / 2.75f)) {
126 const float tTemp = t - (1.5f / 2.75f);
127 return 7.5625f * tTemp * tTemp + 0.75f;
128 } else if (t < (2.5f / 2.75f)) {
129 const float tTemp = t - (2.25f / 2.75f);
130 return 7.5625f * tTemp * tTemp + 0.9375f;
131 } else {
132 const float tTemp = t - (2.625f / 2.75f);
133 return 7.5625f * tTemp * tTemp + 0.984375f;
134 }
135}

137static inline float EaseInBounce(float t) { return 1.0f - EaseOutBounce(1.0f - t); }

139static inline float EaseInOutBounce(float t) {
140 if (t < 0.5f) return 0.5f * EaseInBounce(t * 2.0f);
141 return 0.5f * EaseOutBounce(t * 2.0f - 1.0f) + 0.5f;
